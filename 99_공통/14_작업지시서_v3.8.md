## 작업 목표

generator.py 실행이 끝날 때마다 자동으로 Regression Check를 수행하고
결과를 터미널에 출력 + `Regression_Report.md`로 저장한다.

### Regression 4계층 구조

```
Core Regression      — Generator 공통. 항상 검사.
  ✓ guide.md 읽기
  ✓ rule.md 읽기
  ✓ faq.md 읽기
  ✓ SOP HTML 생성
  ✓ SOP MD 생성
  ✓ Tutorial HTML 생성

Renderer Regression  — Markdown 렌더링. 항상 검사.
  ✓ 코드블록 (<pre><code>)
  ✓ 수평선 (<hr>)
  ✓ H4 (<h4>)
  ✓ 인라인 코드 (<code>)

App Regression       — 앱별 특화. 앱에 따라 항목 다름.
  PrintFlow    : Sticky Tab, PDF 버튼, 탭 구조 (Guide/Rule/FAQ)
  KingTestMaker: PDF 버튼, 탭 구조
  Dooly        : PDF 버튼, 탭 구조

Overall              — Core + Renderer + App 전체 PASS 여부
```

### 설계 원칙
- Generator는 공통 기능(Core + Renderer)만 책임진다
- App별 특화 검증은 App Regression으로 분리하여 앱이 늘어도 구조가 깔끔하게 유지된다
- Report는 계층별로 결과를 구분하여 어디가 실패했는지 즉시 파악 가능하다
- Report 저장 위치: `apps/{App}/output/Regression_Report.md`
- 형식: Markdown (SSOT 철학 일치, Obsidian·Dooly·RAG 호환)

## STEP 2. run_regression_check() 함수 작성

generator.py에 아래 함수를 추가한다.
위치: 파일 하단, `if __name__ == "__main__":` 블록 바로 위.

```python
import os
import datetime

def run_regression_check(app_name, output_dir, input_dir, generated_files):
    """
    4계층 Regression Check 수행 후 터미널 출력 + Regression_Report.md 저장.

    app_name       : 앱 이름 (예: "PrintFlow")
    output_dir     : 출력 폴더 경로
    input_dir      : 입력 폴더 경로 (guide.md 등 위치)
    generated_files: {
        'sop_html':      경로 또는 None,
        'sop_md':        경로 또는 None,
        'tutorial_html': 경로 또는 None,
    }
    """
    now = datetime.datetime.now().strftime('%Y-%m-%d %H:%M')

    # ── 입력 파일 수정 시각 수집 ─────────────────────────────
    input_files = ['guide.md', 'rule.md', 'faq.md']
    input_mtimes = {}
    for fname in input_files:
        fpath = os.path.join(input_dir, fname)
        if os.path.exists(fpath):
            mtime = datetime.datetime.fromtimestamp(
                os.path.getmtime(fpath)
            ).strftime('%Y-%m-%d %H:%M')
            input_mtimes[fname] = mtime
        else:
            input_mtimes[fname] = '파일 없음'

    # ── 체크 헬퍼 ────────────────────────────────────────────
    def check(results, fails, label, condition, fail_msg=None):
        if condition:
            results.append(f'✓ {label}')
        else:
            results.append(f'✗ {label}')
            fails.append(fail_msg or label)

    # ── 1. Core Regression ───────────────────────────────────
    core_results, core_fails = [], []
    sop_html      = generated_files.get('sop_html')
    sop_md        = generated_files.get('sop_md')
    tutorial_html = generated_files.get('tutorial_html')

    check(core_results, core_fails, 'guide.md 읽기',
          input_mtimes['guide.md'] != '파일 없음', 'guide.md 없음')
    check(core_results, core_fails, 'rule.md 읽기',
          input_mtimes['rule.md']  != '파일 없음', 'rule.md 없음')
    check(core_results, core_fails, 'faq.md 읽기',
          input_mtimes['faq.md']   != '파일 없음', 'faq.md 없음')
    check(core_results, core_fails, 'SOP HTML 생성',
          sop_html and os.path.exists(sop_html), 'SOP HTML 생성 실패')
    check(core_results, core_fails, 'SOP MD 생성',
          sop_md   and os.path.exists(sop_md),   'SOP MD 생성 실패')
    check(core_results, core_fails, 'Tutorial HTML 생성',
          tutorial_html and os.path.exists(tutorial_html), 'Tutorial HTML 생성 실패')

    core_pass = not core_fails

    # ── 2. Renderer Regression ───────────────────────────────
    rend_results, rend_fails = [], []
    if sop_html and os.path.exists(sop_html):
        html = open(sop_html, encoding='utf-8').read()
        check(rend_results, rend_fails, '코드블록 렌더링 (<pre><code>)',
              '<pre><code>' in html, 'SOP에 <pre><code> 없음')
        check(rend_results, rend_fails, '수평선 렌더링 (<hr>)',
              '<hr '        in html, 'SOP에 <hr> 없음')
        check(rend_results, rend_fails, 'H4 렌더링 (<h4>)',
              '<h4'         in html, 'SOP에 <h4> 없음')
        check(rend_results, rend_fails, '인라인 코드 렌더링 (<code>)',
              '<code>'      in html, 'SOP에 인라인 <code> 없음')
    else:
        rend_results.append('- SOP HTML 없어 Renderer 검사 건너뜀')

    rend_pass = not rend_fails

    # ── 3. App Regression ────────────────────────────────────
    app_results, app_fails = [], []

    APP_CHECKS = {
        'PrintFlow': [
            ('Sticky Tab',         lambda h: 'sticky-header' in h,   'sticky-header 구조 없음'),
            ('PDF 버튼',           lambda h: 'window.print'  in h,   'PDF 버튼 없음'),
            ('탭 구조 (Guide)',    lambda h: 'id="tab-guide"' in h,  'tab-guide 없음'),
            ('탭 구조 (Rule)',     lambda h: 'id="tab-rule"'  in h,  'tab-rule 없음'),
            ('탭 구조 (FAQ)',      lambda h: 'id="tab-faq"'   in h,  'tab-faq 없음'),
        ],
        'KingTestMaker': [
            ('PDF 버튼',           lambda h: 'window.print'  in h,   'PDF 버튼 없음'),
            ('탭 구조 (Guide)',    lambda h: 'id="tab-guide"' in h,  'tab-guide 없음'),
            ('탭 구조 (Rule)',     lambda h: 'id="tab-rule"'  in h,  'tab-rule 없음'),
            ('탭 구조 (FAQ)',      lambda h: 'id="tab-faq"'   in h,  'tab-faq 없음'),
        ],
        'Dooly': [
            ('PDF 버튼',           lambda h: 'window.print'  in h,   'PDF 버튼 없음'),
            ('탭 구조 (Guide)',    lambda h: 'id="tab-guide"' in h,  'tab-guide 없음'),
            ('탭 구조 (Rule)',     lambda h: 'id="tab-rule"'  in h,  'tab-rule 없음'),
            ('탭 구조 (FAQ)',      lambda h: 'id="tab-faq"'   in h,  'tab-faq 없음'),
        ],
    }

    app_spec = APP_CHECKS.get(app_name, [])
    if app_spec and sop_html and os.path.exists(sop_html):
        html = open(sop_html, encoding='utf-8').read()
        for label, fn, fail_msg in app_spec:
            check(app_results, app_fails, label, fn(html), fail_msg)
    elif not app_spec:
        app_results.append(f'- {app_name} 앱 전용 검사 항목 미정의 (건너뜀)')
    else:
        app_results.append('- SOP HTML 없어 App 검사 건너뜀')

    app_pass = not app_fails

    # ── 4. Overall ───────────────────────────────────────────
    overall = 'PASS' if (core_pass and rend_pass and app_pass) else 'FAIL'

    # ── 다음 권장 작업 ───────────────────────────────────────
    if overall == 'PASS':
        next_tasks = [
            '□ Phase A-3: PDF 전체 출력 개선 (14_작업지시서_v3.9)',
            '□ Phase B:   PrintFlow 실사용 검증',
            '□ Phase B:   KingTestMaker 실제 가이드북 생성',
            '         python analyzer.py KingTestMaker "C:\\Obsidian\\복테메이커" "docs,electron,html,log"',
            '□ Phase B:   Dooly 실제 가이드북 생성',
            '         python analyzer.py Dooly "C:\\Obsidian\\Dooly" "00_System,01_Project,docs,api"',
        ]
    else:
        next_tasks = [
            '□ 위 FAIL 항목 원인 확인 후 재생성',
            '□ generator.py 오류 수정 후 재실행',
        ]

    # ── 터미널 출력 ──────────────────────────────────────────
    sep = '=' * 42
    def section(title, results, passed):
        mark = 'PASS' if passed else 'FAIL'
        print(f'\n  [{title}] {mark}')
        for r in results:
            print(f'    {r}')

    print(f'\n{sep}')
    print(f'  Regression Check  —  {app_name}')
    print(sep)
    section('Core',     core_results, core_pass)
    section('Renderer', rend_results, rend_pass)
    section('App',      app_results,  app_pass)
    print(f'\n{sep}')
    print(f'  Overall : {overall}')
    if overall == 'FAIL':
        all_fails = core_fails + rend_fails + app_fails
        for f in all_fails:
            print(f'    - {f}')
    report_path = os.path.join(output_dir, 'Regression_Report.md')
    print(f'  Report  : {report_path}')
    print(f'{sep}\n')

    # ── Regression_Report.md 생성 ────────────────────────────
    def result_block(title, results, passed):
        mark = '✅ PASS' if passed else '❌ FAIL'
        lines = [f'## {title}  {mark}', '']
        lines += results
        lines.append('')
        return lines

    md = []
    md += ['---']
    md += [f'Project: GuideBook Generator']
    md += [f'App: {app_name}']
    md += [f'Generator: v3.8']
    md += [f'Result: {overall}']
    md += [f'Generated: {now}']
    md += ['']
    md += ['Inputs:']
    for fname, mtime in input_mtimes.items():
        md += [f'  {fname:<10}: {mtime}']
    md += ['---', '']
    md += [f'# Regression Report', '']
    md += [f'생성일: {now}  |  Generator: v3.8  |  App: {app_name}', '']
    md += ['---', '']
    md += result_block('Core',     core_results, core_pass)
    md += ['---', '']
    md += result_block('Renderer', rend_results, rend_pass)
    md += ['---', '']
    md += result_block('App',      app_results,  app_pass)
    md += ['---', '']
    overall_mark = '✅ PASS' if overall == 'PASS' else '❌ FAIL'
    md += [f'## Overall  {overall_mark}', '']
    if overall == 'FAIL':
        md += ['### 실패 원인', '']
        for f in (core_fails + rend_fails + app_fails):
            md += [f'- {f}']
        md += ['']
    md += ['---', '']
    md += ['## 다음 권장 작업', '']
    md += next_tasks
    md += ['']

    with open(report_path, 'w', encoding='utf-8') as f:
        f.write('\n'.join(md))

    return overall
```
