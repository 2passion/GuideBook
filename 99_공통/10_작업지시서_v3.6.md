### 2-1. SOP의 섹션 분류 로직 추가

guide.md에서 섹션을 읽을 때 아래 키워드가 포함된 H2 섹션은
가이드 탭이 아닌 FAQ 탭으로 자동 분류한다.

```python
FAQ_KEYWORDS = ['자주 하는 실수', 'FAQ', '자주묻는', '오류', '에러', '트러블']
RULE_KEYWORDS = ['업무 규칙', '분류 코드', '폴더 구조', '파일명 규칙',
                 '저장 원칙', '검사 규칙', '교재 코드', '과정 코드']
```

섹션 분류 함수:

```python
def classify_sections(content):
    """
    마크다운 content를 H1/H2 섹션 단위로 파싱하여
    guide_body / rule_body / faq_body 세 파트로 분류한다.
    """
    import re
    lines = content.split('\n')
    guide_lines, rule_lines, faq_lines = [], [], []
    current_target = guide_lines
    current_h1 = None

    for line in lines:
        # H1: 새 파트 시작 — 제목 키워드로 분류
        if line.startswith('# ') and not line.startswith('## '):
            title = line[2:].strip()
            if any(k in title for k in ['업무 규칙', '규칙']):
                current_target = rule_lines
                current_h1 = 'rule'
            elif any(k in title for k in ['FAQ']):
                current_target = faq_lines
                current_h1 = 'faq'
            else:
                current_target = guide_lines
                current_h1 = 'guide'
            current_target.append(line)

        # H2: 섹션 키워드로 분류
        elif line.startswith('## '):
            title = line[3:].strip()
            if any(k in title for k in FAQ_KEYWORDS):
                current_target = faq_lines
            elif any(k in title for k in RULE_KEYWORDS):
                current_target = rule_lines
            else:
                # H1 컨텍스트 유지
                if current_h1 == 'rule':
                    current_target = rule_lines
                elif current_h1 == 'faq':
                    current_target = faq_lines
                else:
                    current_target = guide_lines
            current_target.append(line)

        else:
            current_target.append(line)

    return (
        '\n'.join(guide_lines),
        '\n'.join(rule_lines),
        '\n'.join(faq_lines)
    )
```

#### JavaScript 추가 (</body> 앞에 삽입)

```javascript
// ── 탭 전환 ──────────────────────────────────────────────
function switchTab(name, btn) {
  document.querySelectorAll('.tab-panel').forEach(function(p) {
    p.classList.remove('active');
  });
  document.querySelectorAll('.tab-btn').forEach(function(b) {
    b.classList.remove('active');
  });
  document.getElementById('tab-' + name).classList.add('active');
  btn.classList.add('active');
}

// ── 자료 분류 코드 편집 ──────────────────────────────────
var defaultCodes = [
  {code:'01', desc:'교재 원본 자료'},
  {code:'02', desc:'복습 테스트(복테)'},
  {code:'03', desc:'학생별 오답노트'},
  {code:'04', desc:'숙제 출력 자료'},
  {code:'05', desc:'테스트 출력 자료'},
  {code:'06', desc:'내신 대비 모의고사'},
  {code:'07', desc:'기출문제 자료'},
  {code:'08', desc:'진단평가'},
  {code:'09', desc:'입학테스트'}
];

window.addEventListener('DOMContentLoaded', function() {
  loadCodes();
  loadPaths();
});

function loadCodes() {
  var saved = localStorage.getItem('classifyCodes');
  var codes = saved ? JSON.parse(saved) : defaultCodes;
  renderCodes(codes);
}

function renderCodes(codes) {
  var container = document.getElementById('codeRows');
  if (!container) return;
  container.innerHTML = '';
  codes.forEach(function(item) {
    var row = document.createElement('div');
    row.className = 'code-row';
    row.innerHTML =
      '<input class="code-input" value="' + escHtml(item.code) +
      '" placeholder="코드">' +
      '<input class="desc-input" value="' + escHtml(item.desc) +
      '" placeholder="용도 설명">' +
      '<button class="del-btn" onclick="this.parentElement.remove()"' +
      ' title="삭제">✕</button>';
    container.appendChild(row);
  });
}

function escHtml(s) {
  return String(s)
    .replace(/&/g,'&amp;').replace(/"/g,'&quot;')
    .replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function addCodeRow() {
  var container = document.getElementById('codeRows');
  var row = document.createElement('div');
  row.className = 'code-row';
  row.innerHTML =
    '<input class="code-input" value="" placeholder="코드">' +
    '<input class="desc-input" value="" placeholder="용도 설명">' +
    '<button class="del-btn" onclick="this.parentElement.remove()"' +
    ' title="삭제">✕</button>';
  container.appendChild(row);
  container.lastElementChild.querySelector('.code-input').focus();
}

function saveCodes() {
  var rows = document.querySelectorAll('#codeRows .code-row');
  var codes = [];
  rows.forEach(function(row) {
    var inputs = row.querySelectorAll('input');
    var code = inputs[0].value.trim();
    var desc = inputs[1].value.trim();
    if (code && desc) codes.push({code: code, desc: desc});
  });
  localStorage.setItem('classifyCodes', JSON.stringify(codes));
  alert('✅ 자료 분류 코드가 저장되었습니다. (' + codes.length + '개)');
}

// ── 폴더 경로 저장 ───────────────────────────────────────
function loadPaths() {
  var db   = localStorage.getItem('pathDB');
  var sync = localStorage.getItem('pathSync');
  var elDB   = document.getElementById('pathDB');
  var elSync = document.getElementById('pathSync');
  if (db   && elDB)   elDB.value   = db;
  if (sync && elSync) elSync.value = sync;
}

function savePaths() {
  var db   = document.getElementById('pathDB').value.trim();
  var sync = document.getElementById('pathSync').value.trim();
  if (db)   localStorage.setItem('pathDB',   db);
  if (sync) localStorage.setItem('pathSync', sync);
  var status = document.getElementById('pathStatus');
  if (status) {
    status.style.display = 'block';
    setTimeout(function() { status.style.display = 'none'; }, 2500);
  }
}
```
