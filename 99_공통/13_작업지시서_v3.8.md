## STEP 2. run_regression_check() 함수 작성

generator.py에 아래 함수를 추가한다.
위치: 파일 하단, `if __name__ == "__main__":` 블록 바로 위.

```python
import os
import datetime

def run_regression_check(app_name, output_dir, generated_files):
    """
    생성된 출력 파일을 검사하고 Regression_Report.md를 저장한다.

    generated_files: dict
        {
          'sop_html':      경로 또는 None,
          'sop_md':        경로 또는 None,
          'tutorial_html': 경로 또는 None,
        }
    """
    now = datetime.datetime.now().strftime('%Y-%m-%d %H:%M')
    results = []
    fail_reasons = []

    # ── 입력 파일 수정 시각 수집 ──────────────────────────────
    input_dir = os.path.join(os.path.dirname(output_dir), 'input')
    input_files = ['guide.md', 'rule.md', 'faq.md']
    input_mtimes = {}
    for fname in input_files:
        fpath = os.path.join(input_dir, fname)
        if os.path.exists(fpath):
            mtime = datetime.datetime.fromtimestamp(
                os.path.getmtime(fpath)
            ).strftime('%Y-%m-%d %H:%M')
            input_mtimes[fname] = mtime
        else:
            input_mtimes[fname] = '파일 없음'

    # ── 체크 항목 정의 ────────────────────────────────────────
    def check(label, condition, fail_msg=None):
        if condition:
            results.append(f'✓ {label}')
        else:
            results.append(f'✗ {label}')
            fail_reasons.append(fail_msg or label)

    # 입력 파일 존재 여부
    check('guide.md 읽기',  input_mtimes['guide.md'] != '파일 없음', 'guide.md 없음')
    check('rule.md 읽기',   input_mtimes['rule.md']  != '파일 없음', 'rule.md 없음')
    check('faq.md 읽기',    input_mtimes['faq.md']   != '파일 없음', 'faq.md 없음')

    # 출력 파일 존재 여부
    sop_html      = generated_files.get('sop_html')
    sop_md        = generated_files.get('sop_md')
    tutorial_html = generated_files.get('tutorial_html')

    check('SOP HTML 생성',      sop_html      and os.path.exists(sop_html),      'SOP HTML 생성 실패')
    check('SOP MD 생성',        sop_md        and os.path.exists(sop_md),        'SOP MD 생성 실패')
    check('Tutorial HTML 생성', tutorial_html and os.path.exists(tutorial_html), 'Tutorial HTML 생성 실패')

    # SOP HTML 내용 검사
    if sop_html and os.path.exists(sop_html):
        html = open(sop_html, encoding='utf-8').read()
        check('코드블록 렌더링',  '<pre><code>' in html,           'SOP에 <pre><code> 없음')
        check('hr 렌더링',        '<hr '         in html,           'SOP에 <hr> 없음')
        check('Sticky Tab',       'sticky-header' in html,          'sticky-header 구조 없음')
        check('탭 구조 (Guide)',  'id="tab-guide"' in html,         'tab-guide 없음')
        check('탭 구조 (Rule)',   'id="tab-rule"'  in html,         'tab-rule 없음')
        check('탭 구조 (FAQ)',    'id="tab-faq"'   in html,         'tab-faq 없음')
        check('PDF 버튼',         'window.print'   in html,         'PDF 버튼 없음')
    else:
        for label in ['코드블록 렌더링','hr 렌더링','Sticky Tab',
                      '탭 구조 (Guide)','탭 구조 (Rule)','탭 구조 (FAQ)','PDF 버튼']:
            results.append(f'- {label} (SOP HTML 없어 건너뜀)')

    # Tutorial HTML 내용 검사
    if tutorial_html and os.path.exists(tutorial_html):
        thtml = open(tutorial_html, encoding='utf-8').read()
        check('Tutorial STEP 구조', 'step' in thtml.lower(), 'Tutorial STEP 없음')
        check('Tutorial 이전/다음 버튼', ('이전' in thtml or 'prev' in thtml.lower()) and
                                         ('다음' in thtml or 'next' in thtml.lower()),
              'Tutorial 이전/다음 버튼 없음')
    else:
        for label in ['Tutorial STEP 구조', 'Tutorial 이전/다음 버튼']:
            results.append(f'- {label} (Tutorial HTML 없어 건너뜀)')

    # ── 판정 ─────────────────────────────────────────────────
    overall = 'PASS' if not fail_reasons else 'FAIL'

    # ── 다음 권장 작업 결정 ──────────────────────────────────
    next_tasks = []
    if overall == 'PASS':
        next_tasks = [
            '□ Phase A-3: PDF 전체 출력 개선 (14_작업지시서_v3.9)',
            '□ Phase B:   KingTestMaker 실제 가이드북 생성',
            '         python analyzer.py KingTestMaker "C:\\Obsidian\\복테메이커" "docs,electron,html,log"',
            '□ Phase B:   Dooly 실제 가이드북 생성',
            '         python analyzer.py Dooly "C:\\Obsidian\\Dooly" "00_System,01_Project,docs,api"',
        ]
    else:
        next_tasks = [
            '□ 위 FAIL 항목 원인 확인 후 재생성',
            '□ generator.py 오류 수정 후 재실행',
        ]

    # ── 터미널 출력 ──────────────────────────────────────────
    sep = '=' * 40
    print(f'\n{sep}')
    print('Regression Check')
    print(sep)
    for r in results:
        print(f'  {r}')
    print(sep)
    print(f'  결과: {overall}')
    if fail_reasons:
        print('  실패 원인:')
        for f in fail_reasons:
            print(f'    - {f}')
    print(sep)
    print(f'  Report: {os.path.join(output_dir, "Regression_Report.md")}')
    print(sep + '\n')

    # ── Regression_Report.md 생성 ────────────────────────────
    lines = []
    lines.append('---')
    lines.append(f'Project: GuideBook Generator')
    lines.append(f'App: {app_name}')
    lines.append(f'Generator: v3.8')
    lines.append(f'Result: {overall}')
    lines.append(f'Generated: {now}')
    lines.append('')
    lines.append('Inputs:')
    for fname, mtime in input_mtimes.items():
        lines.append(f'  {fname:<10}: {mtime}')
    lines.append('---')
    lines.append('')
    lines.append(f'# Regression Report')
    lines.append('')
    lines.append(f'생성일: {now}  |  Generator: v3.8  |  App: {app_name}')
    lines.append('')
    lines.append('---')
    lines.append('')
    lines.append('## 결과')
    lines.append('')
    for r in results:
        lines.append(r)
    lines.append('')
    lines.append('---')
    lines.append('')
    if overall == 'PASS':
        lines.append('## 최종 판정: ✅ PASS')
        lines.append('')
        lines.append('오류 없음')
    else:
        lines.append('## 최종 판정: ❌ FAIL')
        lines.append('')
        lines.append('### 실패 원인')
        for f in fail_reasons:
            lines.append(f'- {f}')
    lines.append('')
    lines.append('---')
    lines.append('')
    lines.append('## 다음 권장 작업')
    lines.append('')
    for t in next_tasks:
        lines.append(t)
    lines.append('')

    report_path = os.path.join(output_dir, 'Regression_Report.md')
    with open(report_path, 'w', encoding='utf-8') as f:
        f.write('\n'.join(lines))

    return overall
```
