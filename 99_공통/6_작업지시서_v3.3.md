## 수정 목표

generate_tutorial()이 생성하는 HTML의 각 STEP 화면 영역(.screen-area)을
드래그앤드롭 + 파일선택 버튼으로 이미지를 추가할 수 있는 업로드 존으로 변경.

이미지는 localStorage에 base64로 저장되어 브라우저를 닫았다 열어도 유지됨.
이미지가 있으면 해당 STEP에 표시, 없으면 업로드 안내 표시.

## 수정 내용 — generate_tutorial() 의 img_tag() 함수 및 HTML 구조 변경

### 변경 전
img_tag(i) 함수가 이미지 파일이 있으면 base64로 읽어서 img 태그 반환,
없으면 "이미지를 추가하면 여기에 표시됩니다" 안내 div 반환.

### 변경 후
이미지 파일 유무와 관계없이 항상 업로드 가능한 구조로 변경.

각 STEP의 screen-area를 아래 구조로 교체:

<div class="screen-area" id="screen-{prefix}{i}">
  <!-- 이미지가 있을 때 표시 -->
  <div class="img-preview" id="preview-{prefix}{i}" style="display:none;">
    <img id="img-{prefix}{i}" style="max-width:100%;border-radius:8px;
         border:1px solid #e5e7eb;cursor:pointer;" 
         title="클릭하면 이미지를 변경할 수 있습니다">
    <button class="img-delete" onclick="deleteImg('{prefix}',{i})"
            title="이미지 삭제">✕</button>
  </div>
  <!-- 이미지가 없을 때 표시 (업로드 존) -->
  <div class="upload-zone" id="zone-{prefix}{i}"
       onclick="triggerUpload('{prefix}',{i})"
       ondragover="event.preventDefault();this.classList.add('drag-over')"
       ondragleave="this.classList.remove('drag-over')"
       ondrop="handleDrop(event,'{prefix}',{i})">
    <div class="upload-icon">📸</div>
    <div class="upload-text">
      클릭하거나 이미지를 여기에 드래그하세요<br>
      <small>JPG, PNG, WebP 지원</small>
    </div>
  </div>
  <input type="file" id="file-{prefix}{i}" accept="image/*"
         style="display:none"
         onchange="handleFile(event,'{prefix}',{i})">
</div>

### JavaScript 추가 (튜토리얼 HTML의 <script> 안에 추가)

// ── 이미지 업로드 관련 ──────────────────────────

// 페이지 로드 시 저장된 이미지 복원
window.addEventListener('DOMContentLoaded', function() {
  // localStorage에서 이미지 복원
  for (var key in localStorage) {
    if (key.startsWith('img_')) {
      var parts = key.replace('img_', '').split('_');
      var prefix = parts[0];
      var idx = parseInt(parts[1]);
      var dataUrl = localStorage.getItem(key);
      if (dataUrl) showImg(prefix, idx, dataUrl);
    }
  }
});

function triggerUpload(prefix, idx) {
  document.getElementById('file-' + prefix + idx).click();
}

function handleFile(event, prefix, idx) {
  var file = event.target.files[0];
  if (!file) return;
  readAndSave(file, prefix, idx);
}

function handleDrop(event, prefix, idx) {
  event.preventDefault();
  document.getElementById('zone-' + prefix + idx)
          .classList.remove('drag-over');
  var file = event.dataTransfer.files[0];
  if (!file || !file.type.startsWith('image/')) return;
  readAndSave(file, prefix, idx);
}

function readAndSave(file, prefix, idx) {
  var reader = new FileReader();
  reader.onload = function(e) {
    var dataUrl = e.target.result;
    // localStorage에 저장
    localStorage.setItem('img_' + prefix + '_' + idx, dataUrl);
    showImg(prefix, idx, dataUrl);
  };
  reader.readAsDataURL(file);
}

function showImg(prefix, idx, dataUrl) {
  var img = document.getElementById('img-' + prefix + idx);
  var preview = document.getElementById('preview-' + prefix + idx);
  var zone = document.getElementById('zone-' + prefix + idx);
  if (!img || !preview || !zone) return;
  img.src = dataUrl;
  // 이미지 클릭 시 파일 선택
  img.onclick = function() { triggerUpload(prefix, idx); };
  preview.style.display = 'flex';
  zone.style.display = 'none';
}

function deleteImg(prefix, idx) {
  localStorage.removeItem('img_' + prefix + '_' + idx);
  var img = document.getElementById('img-' + prefix + idx);
  var preview = document.getElementById('preview-' + prefix + idx);
  var zone = document.getElementById('zone-' + prefix + idx);
  if (!img || !preview || !zone) return;
  img.src = '';
  preview.style.display = 'none';
  zone.style.display = 'flex';
  // input 초기화
  var fileInput = document.getElementById('file-' + prefix + idx);
  if (fileInput) fileInput.value = '';
}

// ── 전체 이미지 초기화 버튼용 ──────────────────
function clearAllImages() {
  if (!confirm('모든 STEP의 이미지를 삭제하시겠습니까?')) return;
  var keys = [];
  for (var key in localStorage) {
    if (key.startsWith('img_')) keys.push(key);
  }
  keys.forEach(function(k) {
    var parts = k.replace('img_', '').split('_');
    var prefix = parts[0];
    var idx = parseInt(parts[1]);
    deleteImg(prefix, idx);
    localStorage.removeItem(k);
  });
}

## 사용 방법 (완성 후 신규 조교 안내용)

1. 튜토리얼 HTML을 브라우저로 열기
2. 각 STEP의 📸 영역에 스크린샷을 드래그앤드롭
   또는 클릭 후 파일 선택
3. 이미지는 자동으로 저장됨 (같은 브라우저에서만 유지)
4. 이미지 변경: 기존 이미지 클릭 → 새 파일 선택
5. 이미지 삭제: ✕ 버튼 클릭

> ⚠️ localStorage는 같은 PC·같은 브라우저에서만 유지됩니다.
> 다른 PC에서도 이미지가 필요하다면 generator.py를 재실행하는 방식B를 사용하세요.
